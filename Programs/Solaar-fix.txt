if nothing happens with key diversion, run this:
# 1) Install the Wayland/uinput-friendly Solaar rule (overrides the package default)
sudo curl -o /etc/udev/rules.d/42-logitech-unify-permissions.rules \
  https://raw.githubusercontent.com/pwr-Solaar/Solaar/master/rules.d-uinput/42-logitech-unify-permissions.rules

# 2) Reload udev and apply rules
sudo udevadm control --reload-rules
sudo udevadm trigger

# 3) Ensure the uinput kernel module is loaded
lsmod | grep -q uinput || sudo modprobe uinput

# 4) Unplug/replug the receiver (or reboot)

# If it still does not work, try
sudo modprobe uinput
echo uinput | sudo tee /etc/modules-load.d/uinput.conf
